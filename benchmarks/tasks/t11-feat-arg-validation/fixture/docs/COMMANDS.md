# Commands

## add

Usage: relay add <args>
Args: a -- first integer
Args: b -- second integer

## count

Usage: relay count <args>
Args: text -- string to measure

## echo

Usage: relay echo <args>
Args: text -- string to echo back

## head

Usage: relay head <args>
Args: text -- input string
Args: n -- number of characters to return

## join

Usage: relay join <args>
Args: left -- first string
Args: sep -- separator string
Args: right -- second string

## lower

Usage: relay lower <args>
Args: text -- string to lower-case

## mul

Usage: relay mul <args>
Args: a -- first integer
Args: b -- second integer

## ping

Usage: relay ping
Args: (none)

## repeat

Usage: relay repeat <args>
Args: text -- string to repeat
Args: times -- number of repetitions

## split

Usage: relay split <args>
Args: text -- string to split
Args: sep -- separator string

## tail

Usage: relay tail <args>
Args: text -- input string
Args: n -- number of characters to return from the end

## trim

Usage: relay trim <args>
Args: text -- string to trim

## upper

Usage: relay upper <args>
Args: text -- string to upper-case

## wrap

Usage: relay wrap <args>
Args: text -- string to wrap
Args: prefix -- prefix to prepend
Args: suffix -- suffix to append
